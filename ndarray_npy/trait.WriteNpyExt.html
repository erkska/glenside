Extension trait for writing ArrayBase to .npy files.

Example

use ndarray::{array, Array2};
use ndarray_npy::WriteNpyExt;
use std::fs::File;

let arr: Array2<i32> = array![[1, 2, 3], [4, 5, 6]];
let writer = File::create("array.npy")?;
arr.write_npy(writer)?;
